        <script>
            /**
             * Password Validation Script
             * * */

            /**
             * Creating two variables for password and confirmPassword
             * Getting their values from the user input
             * @Harman - I have changed the const to let. - Jody
             * * */
            const Password_A = document.getElementById('inputPassword1').value;
            const Password_B = document.getElementById('inputPassword2').value;
            const Msg_Element = document.getElementById('inputPassword2');

            /**
             * @function - Check and Validate Passwords
             * * */
            const Validate = (p1, p2) => {
                p1 != p2
                    ? SetValidityMessege('Passwords do not match')
                    : SetValidityMessege('');
            };

            const SetValidityMessege = (messege) => {
                Msg_Element.setCustomValidity(messege);
            };

            /** Event Handlers for input */
            if (Password_A.onchange || Password_B.onkeyup) {
                Validate(Password_A, Password_B);
            }
        </script>
